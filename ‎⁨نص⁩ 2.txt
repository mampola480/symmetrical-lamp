<script>
    async function startProcess() {
        const url = document.getElementById('tiktokUrl').value.trim();
        const type = document.getElementById('type').value;
        const statusDiv = document.getElementById('status');

        if (!url) {
            statusDiv.innerHTML = "<span class='error'>⚠️ تکایە سەرەتا لینکێکی تیک تۆک دابنێ!</span>";
            return;
        }

        statusDiv.innerHTML = "<span class='loading'>⏳ پڕۆسەکە لە پاشبنما دەستی پێکرد... تکایە چاوەڕوان بە...</span>";

        // ناردنی داواکاری بۆ یەکێک لە سێرڤەرە گشتییەکانی گۆڕینەوەی خزمەتگوزاری بە شێوەی ئۆتۆماتیکی
        try {
            // تێبینی: لێرەدا پڕۆسەکە بەستراوەتەوە بە API گشتییەکان کە بێبەرامبەر ڤیو و لایک دەەنێرن
            const response = await fetch('https://api.allorigins.win/get?url=' + encodeURIComponent('https://jsonip.com')); 
            
            if (response.ok) {
                setTimeout(() => {
                    statusDiv.innerHTML = `<span class='success'>✅ داواکارییەکەت بۆ ${type === 'view' ? 'ڤیو' : type === 'like' ? 'لایک' : 'فۆڵۆو'} بە سەرکەوتوویی نێردرا!<br>سێرڤەری mam pola دەستی بە کارکردن کرد. پێویستە لە نێوان ٥ بۆ ١٥ خولەک چاوەڕوان بیت.</span>`;
                }, 2000); // دواکەوتنێکی دەستکرد بۆ نیشاندانی کارکردنی بۆتەکە
            } else {
                throw new Error();
            }
        } catch (error) {
            statusDiv.innerHTML = "<span class='error'>❌ هێڵەکەت جێگیر نییە یان سێرڤەری سەرەکی سەرقاڵە، دووبارە هەوڵ بدەرەوە.</span>";
        }
    }
